#include <Arduino.h>

#define MIC_PIN        34
#define LED_PIN        27
#define LIMIAR         40     // silêncio fica abaixo, palma bem acima
#define LED_DURACAO_MS 500    // ms que o LED fica aceso

unsigned long led_acendeu = 0;

void setup() {
  Serial.begin(115200);
  pinMode(LED_PIN, OUTPUT);
}

void loop() {
  int som = analogRead(MIC_PIN);

  if (som > LIMIAR) {
    led_acendeu = millis();
    Serial.printf("[PALMA] som=%d\n", som);
  }

  digitalWrite(LED_PIN, (millis() - led_acendeu < LED_DURACAO_MS) ? HIGH : LOW);

  Serial.println(som);
  delay(1);
}
